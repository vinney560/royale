"""

## installs qrcode and Pillow.

# packages needed; Pillow and qrcode

import qrcode
from datetime import datetime


class QRGenerator:
    """
    A simple QR code generator with basic customization options.
    
    Example:
        >>> generator = QRGenerator()
        >>> generator.generate("Hello World", "my_qr.png")
    """
    
    def __init__(self, fill_color="black", back_color="white", size=10):
        """
        Set up the QR generator with your preferred colors and size.
        
        Args:
            fill_color: Color of the QR code (default: "black")
            back_color: Background color (default: "white")
            size: How big each pixel block is (default: 10)
        
        Colors can be:
            - Names: "red", "blue", "green", etc.
            - RGB: (255, 0, 0) for red
            - Hex: "#FF0000" for red
        """
        self.fill_color = fill_color
        self.back_color = back_color
        self.size = size
    
    def generate(self, data, filename=None):
        """
        Create a QR code from your data and save it as an image.
        
        Args:
            data: The text/URL you want to encode
            filename: Where to save the image (optional)
                     If not provided, auto-generates a name
        
        Returns:
            The filename of the saved image
        
        Example:
            >>> gen = QRGenerator()
            >>> gen.generate("I Love Coding", "love_qr.png")
            'love_qr.png'
        """
        # Auto-generate filename if not provided
        if filename is None:
            timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
            filename = f"royale-qr_code_{timestamp}.png"
        
        # Create the QR code
        qr = qrcode.QRCode(
            version=1,  # Smallest QR code size. Read Below for more context.
            error_correction=qrcode.constants.ERROR_CORRECT_H,  # High error correction
            box_size=self.size,
            border=4,
        )
        
        # Add your data
        qr.add_data(data)
        qr.make(fit=True)
        
        # Create the image with your colors
        img = qr.make_image(
            fill_color=self.fill_color,
            back_color=self.back_color
        )
        
        # Save it
        img.save(filename)
        print(f"✅ QR Code saved as: {filename}")
        return filename


def quick_qr(data, filename=None):
    """
    The fastest way to generate a QR code with default settings.
    
    Args:
        data: The text/URL to encode
        filename: Where to save (optional)
    
    Example:
        >>> quick_qr("https://example.com", "website.png")
    """
    generator = QRGenerator()
    return generator.generate(data, filename)


# ============================================
# HOW TO USE THIS FILE
# ============================================

if __name__ == "__main__":
    # Example 1: Basic usage with default colors (black on white)
    gen = QRGenerator()
    gen.generate("I Love Coding")
    
    # Example 2: Custom colors
    gen = QRGenerator(fill_color="black", back_color="lightgreen")
    gen.generate("I Love Coding", "green_background.png")
    
    # Example 3: Quick one-liner
    quick_qr("https://example.com", "website_qr.png")
    
    print("\n✨ All QR codes generated successfully!")
